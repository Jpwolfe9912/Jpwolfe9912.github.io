127size_t
128lwrb_write(lwrb_t* buff, const void* data, size_t btw) {
129 size_t tocopy, free;
130 volatile size_t buff_w_ptr;
131 const uint8_t* d = data;
132
133 if (!BUF_IS_VALID(buff) || data == NULL || btw == 0) {
134 return 0;
135 }
136
137 /* Calculate maximum number of bytes available to write */
138 free = lwrb_get_free(buff);
139 btw = BUF_MIN(free, btw);
140 if (btw == 0) {
141 return 0;
142 }
143
144 /* Step 1: Write data to linear part of buffer */
145 buff_w_ptr = buff->w;
146 tocopy = BUF_MIN(buff->size - buff_w_ptr, btw);
147 BUF_MEMCPY(&buff->buff[buff_w_ptr], d, tocopy);
148 buff_w_ptr += tocopy;
149 btw -= tocopy;
150
151 /* Step 2: Write data to beginning of buffer (overflow part) */
152 if (btw > 0) {
153 BUF_MEMCPY(buff->buff, &d[tocopy], btw);
154 buff_w_ptr = btw;
155 }
156
157 /* Step 3: Check end of buffer */
158 if (buff_w_ptr >= buff->size) {
159 buff_w_ptr = 0;
160 }
161
162 /*
163 * Write final value to the actual running variable.
164 * This is to ensure no read operation can access intermediate data
165 */
166 buff->w = buff_w_ptr;
167
168 BUF_SEND_EVT(buff, LWRB_EVT_WRITE, tocopy + btw);
169 return tocopy + btw;
170}

181size_t
182lwrb_read(lwrb_t* buff, void* data, size_t btr) {
183 size_t tocopy, full;
184 volatile size_t buff_r_ptr;
185 uint8_t* d = data;
186
187 if (!BUF_IS_VALID(buff) || data == NULL || btr == 0) {
188 return 0;
189 }
190
191 /* Calculate maximum number of bytes available to read */
192 full = lwrb_get_full(buff);
193 btr = BUF_MIN(full, btr);
194 if (btr == 0) {
195 return 0;
196 }
197
198 /* Step 1: Read data from linear part of buffer */
199 buff_r_ptr = buff->r;
200 tocopy = BUF_MIN(buff->size - buff_r_ptr, btr);
201 BUF_MEMCPY(d, &buff->buff[buff_r_ptr], tocopy);
202 buff_r_ptr += tocopy;
203 btr -= tocopy;
204
205 /* Step 2: Read data from beginning of buffer (overflow part) */
206 if (btr > 0) {
207 BUF_MEMCPY(&d[tocopy], buff->buff, btr);
208 buff_r_ptr = btr;
209 }
210
211 /* Step 3: Check end of buffer */
212 if (buff_r_ptr >= buff->size) {
213 buff_r_ptr = 0;
214 }
215
216 /*
217 * Write final value to the actual running variable.
218 * This is to ensure no write operation can access intermediate data
219 */
220 buff->r = buff_r_ptr;
221
222 BUF_SEND_EVT(buff, LWRB_EVT_READ, tocopy + btr);
223 return tocopy + btr;
224}

234size_t
235lwrb_peek(lwrb_t* buff, size_t skip_count, void* data, size_t btp) {
236 size_t full, tocopy;
237 volatile size_t r;
238 uint8_t* d = data;
239
240 if (!BUF_IS_VALID(buff) || data == NULL || btp == 0) {
241 return 0;
242 }
243
244 r = buff->r;
245
246 /* Calculate maximum number of bytes available to read */
247 full = lwrb_get_full(buff);
248
249 /* Skip beginning of buffer */
250 if (skip_count >= full) {
251 return 0;
252 }
253 r += skip_count;
254 full -= skip_count;
255 if (r >= buff->size) {
256 r -= buff->size;
257 }
258
259 /* Check maximum number of bytes available to read after skip */
260 btp = BUF_MIN(full, btp);
261 if (btp == 0) {
262 return 0;
263 }
264
265 /* Step 1: Read data from linear part of buffer */
266 tocopy = BUF_MIN(buff->size - r, btp);
267 BUF_MEMCPY(d, &buff->buff[r], tocopy);
268 btp -= tocopy;
269
270 /* Step 2: Read data from beginning of buffer (overflow part) */
271 if (btp > 0) {
272 BUF_MEMCPY(&d[tocopy], buff->buff, btp);
273 }
274 return tocopy + btp;
275}

282size_t
283lwrb_get_free(lwrb_t* buff) {
284 size_t size;
285 volatile size_t w, r;
286
287 if (!BUF_IS_VALID(buff)) {
288 return 0;
289 }
290
291 /* Use temporary values in case they are changed during operations */
292 w = buff->w;
293 r = buff->r;
294 if (w == r) {
295 size = buff->size;
296 } else if (r > w) {
297 size = r - w;
298 } else {
299 size = buff->size - (w - r);
300 }
301
302 /* Buffer free size is always 1 less than actual size */
303 return size - 1;
304}

398lwrb_skip(lwrb_t* buff, size_t len) {
399 size_t full;
400 volatile size_t r;
401
402 if (!BUF_IS_VALID(buff) || len == 0) {
403 return 0;
404 }
405
406 full = lwrb_get_full(buff);
407 len = BUF_MIN(len, full);
408 r = buff->r + len;

435size_t
436lwrb_get_linear_block_write_length(lwrb_t* buff) {
437 size_t len;
438 volatile size_t w, r;
439
440 if (!BUF_IS_VALID(buff)) {
441 return 0;
442 }
443
444 /* Use temporary values in case they are changed during operations */
445 w = buff->w;
446 r = buff->r;
447 if (w >= r) {
448 len = buff->size - w;
449 /*
450 * When read pointer is 0,
451 * maximal length is one less as if too many bytes
452 * are written, buffer would be considered empty again (r == w)
453 */
454 if (r == 0) {
455 /*
456 * Cannot overflow:
457 * - If r is not 0, statement does not get called
458 * - buff->size cannot be 0 and if r is 0, len is greater 0
459 */
460 --len;
461 }
462 } else {
463 len = r - w - 1;
464 }
465 return len;
466}
