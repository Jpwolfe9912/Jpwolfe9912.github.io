154spiReadBytes(uint8_t reg, uint8_t *pData, uint8_t size)
155{
156 reg = reg | 0x80; // read operation
157
158 DMA2_Stream0->CR &= ~DMA_SxCR_EN;
159 while(DMA2_Stream0->CR & DMA_SxCR_EN);
160 DMA2_Stream0->CR |= (0x3 << 25U);
161 DMA2_Stream0->NDTR = size;
162 DMA2_Stream0->M0AR = (uint32_t)pData;
163
164 DMA2_Stream3->CR &= ~DMA_SxCR_EN;
165 while(DMA2_Stream3->CR & DMA_SxCR_EN);
166 DMA2_Stream3->CR |= (0x3 << 25U);
167 DMA2_Stream3->NDTR = size;
168 DMA2_Stream3->M0AR = (uint32_t)&reg;
169
170 SPI1_ENABLE;
171
172 DMA2_Stream0->CR |= DMA_SxCR_EN;
173 DMA2_Stream3->CR |= DMA_SxCR_EN;
174
175 SPI1->CR2 |= SPI_CR2_RXDMAEN;
176 SPI1->CR2 |= SPI_CR2_TXDMAEN;
177
178 SPI1->CR1 |= SPI_CR1_SPE;
