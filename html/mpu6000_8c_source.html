175void
176computeMPU6000RTData(void)
177{
178 uint8_t axis;
179 uint16_t samples;
180
181 float accelSum[3] = { 0.0f, 0.0f, 0.0f };
182 float gyroSum[3] = { 0.0f, 0.0f, 0.0f };
183
184 mpu6000Calibrating = true;
185
186 for (samples = 0; samples < 5000; samples++)

189

192 accelSum[XAXIS] += ((float)rawAccel[XAXIS].value - eepromConfig.accelBiasMPU[XAXIS] - accelTCBias[XAXIS]) * eepromConfig.accelScaleFactorMPU[XAXIS];
193 accelSum[YAXIS] += ((float)rawAccel[YAXIS].value - eepromConfig.accelBiasMPU[YAXIS] - accelTCBias[YAXIS]) * eepromConfig.accelScaleFactorMPU[YAXIS];
194 accelSum[ZAXIS] += ((float)rawAccel[ZAXIS].value - eepromConfig.accelBiasMPU[ZAXIS] - accelTCBias[ZAXIS]) * eepromConfig.accelScaleFactorMPU[ZAXIS];
195
196 gyroSum[ROLL ] += (float)rawGyro[ROLL ].value - gyroTCBias[ROLL ];
197 gyroSum[PITCH] += (float)rawGyro[PITCH].value - gyroTCBias[PITCH];
198 gyroSum[YAW ] += (float)rawGyro[YAW ].value - gyroTCBias[YAW ];
199
200 delayMicroseconds(100);
201 }
202
203 for (axis = 0; axis < 3; axis++)
204 {
205 accelSum[axis] = accelSum[axis] / 5000.0f;
206 gyroRTBias[axis] = gyroSum[axis] / 5000.0f;
207 }
208
209 accelOneG = sqrt(SQR(accelSum[XAXIS]) + SQR(accelSum[YAXIS]) + SQR(accelSum[ZAXIS]));
210
211 mpu6000Calibrating = false;
212}
