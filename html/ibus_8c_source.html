114 if(b == 0x20){
115
116 frameLength = b;
117 state++;
118 }
119 else
120 {
121 status = IBUS_ERROR;
122 }
123 break;
124 }
125 case IBUS_STATE_TYPE: { // type byte
126 if(b == 0x40){
127 devID = b;
128 state++;
129 }
130 else
131 status = IBUS_ERROR;
132 break;
133 }
134 case IBUS_STATE_PAYLOAD: { // payload bytes
135 ibusPayload[payloadPos++] = b;
136 if(payloadPos == PAYLOAD_SIZE)
137 {
138 ibus_update(ibusPayload);
139 state++;
140 payloadPos = 0;
141 }
142 break;
143 }
144 case IBUS_STATE_CRC: { // crc bytes
145 ibusCRC[crcPos++] = b;
146 if(crcPos == CRC_SIZE){
147 if(ibus_frame_crc(ibusCRC))
148 {
149 status = IBUS_READY;
150 }
151 else{
152 status = IBUS_ERROR;
153 }
154 crcPos = 0;
155 }
156 break;
157 }
158 }
159 if(status == IBUS_ERROR)
160 {
161 framePos = 0;
162 state = 0;
163 }
164 if(status == IBUS_READY)
165 {
166 framePos = 0;
167 state = 0;
168 }
