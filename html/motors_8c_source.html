129 /* Send 0 command 1500 times */
130 motor_value[MOTOR1] = DSHOT_CMD_MOTOR_STOP;
131 motor_value[MOTOR2] = DSHOT_CMD_MOTOR_STOP;
132 motor_value[MOTOR3] = DSHOT_CMD_MOTOR_STOP;
133 motor_value[MOTOR4] = DSHOT_CMD_MOTOR_STOP;
134 dshotWait(DSHOT_ARM_COUNT);
135
136 /* Send nothing for 10ms */
137 motor_initialized = true;
138 delay(10);
139 motor_initialized = false;
140
141 /* Send command 10 times */
142 motor_value[MOTOR1] = ((motors & 0x01) >> 0) * command;
143 motor_value[MOTOR2] = ((motors & 0x02) >> 1) * command;
144 motor_value[MOTOR3] = ((motors & 0x04) >> 2) * command;
145 motor_value[MOTOR4] = ((motors & 0x08) >> 3) * command;
146 dshotWait(DSHOT_SETTINGS_COUNT);
147
148 /* Send 0 command 1500 times */
149 motor_value[MOTOR1] = DSHOT_CMD_MOTOR_STOP;
150 motor_value[MOTOR2] = DSHOT_CMD_MOTOR_STOP;
151 motor_value[MOTOR3] = DSHOT_CMD_MOTOR_STOP;
152 motor_value[MOTOR4] = DSHOT_CMD_MOTOR_STOP;
153 dshotWait(DSHOT_ARM_COUNT);
154
155 /* Change modes */
156 if(command == DSHOT_CMD_3D_MODE_ON)
157 {
158 motors_mode = MOTORS_MODE_3D;
159 printf("\nChanged motors 3D mode");
160 }
161 else if(command == DSHOT_CMD_3D_MODE_OFF)
